**ui/packages/client/src/pages/CatalogPage.tsx** (from RESEARCH.md Pattern 3 + PATTERNS.md CatalogPage section):
```typescript
import React from "react"
import { useQuery } from "@tanstack/react-query"
import type { CatalogApp } from "@librepod/shared"
import { AppCard } from "@/components/AppCard"
import { AppCardSkeleton } from "@/components/AppCardSkeleton"
import { ErrorBlock } from "@/components/ErrorBlock"
import { EmptyState } from "@/components/EmptyState"

const GRID_STYLE: React.CSSProperties = {
  display: "grid",
  gridTemplateColumns: "repeat(auto-fill, minmax(280px, 1fr))",
  gap: 24,
}

export function CatalogPage() {
  const { isPending, isError, data, refetch } = useQuery<CatalogApp[]>({
    queryKey: ["apps"],
    queryFn: async () => {
      const res = await fetch("/api/apps")
      if (!res.ok) throw new Error("Failed to fetch apps")
      return res.json()
    },
    retry: 0,
  })

  if (isPending) {
    return (
      <div style={GRID_STYLE}>
        {Array.from({ length: 12 }).map((_, i) => (
          <AppCardSkeleton key={i} />
        ))}
      </div>
    )
  }

  if (isError) {
    return <ErrorBlock onRetry={refetch} />
  }

  if (!data || data.length === 0) {
    return <EmptyState />
  }

  return (
    <div style={GRID_STYLE}>
      {data.map((app) => (
        <AppCard key={app.name} app={app} />
      ))}
    </div>
  )
}
```

```typescript
import { useParams, Link } from "react-router-dom"
import { useQuery } from "@tanstack/react-query"
import type { CatalogApp } from "@librepod/shared"
import { AppIcon } from "@/components/AppIcon"
import { Badge } from "@/components/ui/badge"
import { Button } from "@/components/ui/button"
import { Skeleton } from "@/components/ui/skeleton"
import { Separator } from "@/components/ui/separator"

function DetailSkeleton() {
  return (
    <div className="mx-auto max-w-2xl">
      <Skeleton className="h-4 w-32 mb-6" />
      <Skeleton className="h-20 w-20 rounded-md mb-4" />
      <Skeleton className="h-8 w-1/2 mb-2" />
      <Skeleton className="h-4 w-1/4 mb-6" />
      <Skeleton className="h-24 w-full" />
    </div>
  )
}

function NotFoundBlock() {
  return (
    <div className="mx-auto max-w-2xl text-center mt-12">
      <h2 className="text-xl font-semibold">App not found</h2>
      <p className="mt-2 text-sm text-muted-foreground">
        This app doesn&apos;t exist in the catalog.
      </p>
      <Link to="/" className="mt-4 inline-block text-sm text-muted-foreground hover:underline">
        ← Back to catalog
      </Link>
    </div>
  )
}

export function AppDetailPage() {
  const { name } = useParams<{ name: string }>()

  const { isPending, isError, error, data } = useQuery<CatalogApp>({
    queryKey: ["apps", name],
    queryFn: async () => {
      const res = await fetch(`/api/apps/${name}`)
      // Check 404 BEFORE !res.ok (RESEARCH.md Pitfall 7)
      if (res.status === 404) throw new Error("NOT_FOUND")
      if (!res.ok) throw new Error("Failed to fetch app")
      return res.json()
    },
    retry: 0,
  })

  if (isPending) return <DetailSkeleton />
  if (isError && (error as Error)?.message === "NOT_FOUND") return <NotFoundBlock />
  if (isError || !data) return <NotFoundBlock />

  return (
    <div className="mx-auto max-w-2xl">
      <Link
        to="/"
        className="text-sm text-muted-foreground hover:underline"
      >
        ← Back to catalog
      </Link>

      <div className="mt-4 rounded-lg border border-border bg-card p-8">
        <AppIcon src={data.icon} name={data.displayName} size={80} />

        <h1 className="mt-4 text-[28px] font-semibold leading-[1.2]">
          {data.displayName}
        </h1>

        <div className="mt-2 flex items-center gap-3">
          <Badge variant="secondary">{data.category}</Badge>
          <span className="text-sm text-muted-foreground">
            Version: {data.version}
          </span>
        </div>

        <Separator className="my-6" />

        <p className="text-sm leading-relaxed">{data.description}</p>

        <div className="mt-4">
          <a
            href={data.sourceUrl}
            target="_blank"
            rel="noopener noreferrer"
            className="text-sm underline text-muted-foreground hover:text-foreground"
          >
            View source
          </a>
        </div>

        <div className="mt-8">
          <Button
            disabled
            title="Install coming soon"
            className="opacity-50 cursor-not-allowed"
          >
            Install App
          </Button>
        </div>
      </div>
    </div>
  )
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| URL :name param → fetch /api/apps/:name | User-controlled URL segment; passed as URL path to backend |
| data.sourceUrl → anchor href | OCI URL rendered in href; opened in new tab with noopener |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-T05 | Tampering | :name param in fetch URL | accept | Backend validates existence and returns 404 for unknown names (Phase 1 implemented); frontend reflects backend response |
| T-02-T06 | Tampering | sourceUrl rendered as href | accept | sourceUrl is an OCI URL from catalog.yaml (trusted backend source); rendered as anchor href only — no eval, no dangerouslySetInnerHTML; rel="noopener noreferrer" prevents opener access |
</threat_model>
