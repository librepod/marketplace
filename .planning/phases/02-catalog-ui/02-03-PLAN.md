**ui/packages/client/src/main.tsx** (exact pattern from RESEARCH.md Code Examples):
```typescript
import React from "react"
import ReactDOM from "react-dom/client"
import { RouterProvider } from "react-router-dom"
import { QueryClient, QueryClientProvider } from "@tanstack/react-query"
import { router } from "./router"
import "@fontsource/inter"
import "./index.css"

// D-02: default dark mode — set BEFORE React mounts to prevent flash of unstyled content
document.documentElement.classList.add("dark")

const queryClient = new QueryClient({
  defaultOptions: {
    queries: {
      staleTime: 5 * 60 * 1000, // 5 minutes
      retry: 0,                  // Manual retry via Retry button; no automatic retries
    },
  },
})

ReactDOM.createRoot(document.getElementById("root")!).render(
  <React.StrictMode>
    <QueryClientProvider client={queryClient}>
      <RouterProvider router={router} />
    </QueryClientProvider>
  </React.StrictMode>
)
```

**ui/packages/client/src/router.tsx** (pattern from RESEARCH.md Pattern 2):
```typescript
import { createBrowserRouter } from "react-router-dom"
import { AppShell } from "./components/AppShell"
import { CatalogPage } from "./pages/CatalogPage"
import { AppDetailPage } from "./pages/AppDetailPage"

export const router = createBrowserRouter([
  {
    element: <AppShell />,
    children: [
      { path: "/", element: <CatalogPage /> },
      { path: "/apps/:name", element: <AppDetailPage /> },
    ],
  },
])
```
Note: Routes are nested under AppShell using the Outlet pattern. AppShell renders <Outlet /> where page content goes.

- `<Outlet />` is where React Router renders the current route's page component
